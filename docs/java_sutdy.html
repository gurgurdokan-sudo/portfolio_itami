public class mainCard {                                                                                                                        
	public static void main(String[] args) {
		int radius = 20; //半径
		int X = radius;
		int Y = radius;
		for (int i = 0; i <= radius * 2; i++) {// 縦
			for (int j = 0; j <= radius * 2; j++) {// 横
				double disSquar = Math.pow(j - X, 2) + Math.pow(i - Y, 2);
				// (x-cx)²+(y-cy)²=r² を再現 中心からの距離を計算してDouble型に入れる
				
				if (disSquar - Math.pow(radius, 2) < radius) {
					// disSquar(中心)-円の2乗がradiusの二倍(円の直径)よりも小さい
					System.out.print(" *");
				} else {
					System.out.print("  ");
				}
			}
			System.out.println();
		}
	}
}

import java.util.ArrayList;
import java.util.List;
import java.util.Scanner;
public class Main1 {
public static void main(String[] args) {
	List BallsBoxList = new ArrayList<>();
	BallsBoxList.add(new Box("赤", "四角形"));
	BallsBoxList.add(new Box("青", "円形"));
	boolean bool = true;
	while (bool) {
		System.out.println("[menu]1. ボール投入 / 2. ボール取出 / 3. ボール総数表示 / 4. 終了>");
		int input = new Scanner(System.in).nextInt();
		switch (input) {
		case 1:
			int[] select = inBalls(BallsBoxList, "投入");
			BallsBoxList.get(select[0]).insertBalls(select[1]);
			System.out.println(select[1] + "個のボールを入れました。");
			break;
		case 2:
			select = inBalls(BallsBoxList, "取り出し");
			if (BallsBoxList.get(select[0]).getNum() > select[1]) {
				BallsBoxList.get(select[0]).setNum(select[1]);
				Box.totalNumber -= select[1];
				System.out.println(select[1] + "個のボールを取り出しました。");
			} else {
				Box.totalNumber -= BallsBoxList.get(select[0]).getNum();
				System.out.println(BallsBoxList.get(select[0]).getNum() + "個のボールを取り出しました。");
				BallsBoxList.get(select[0]).setNum(0);
			}
			break;
		case 3:
			System.out.println(Box.totalNumber + "個のボールがあります。");
			break;
		case 4:
			System.out.println("---------------it over------------");
			bool = false;
			break;
		default:
			System.out.println("入力誤り");
		}
	}
}

public static int[] inBalls(List b, String str) {//Listと投入or取出
	int[] select = new int[2];
	Scanner s = new Scanner(System.in);
	System.out.println("[menu] ");
	for (int i = 0; i < b.size(); i++) {
		System.out.println("|" + i + "|" + b.get(i).toString());
	}
	System.out.println("ボールを" + str + "する箱の番号を入力して下さい > ");
	select[0] = s.nextInt();
	System.out.println("選択された<<" + b.get(select[0]).toString() + ">から");
	System.out.println(str + "するボールの個数を入力して下さい > ");
	select[1] = s.nextInt();

	return select;
}//MainにstaticMethodを定義する事で投入/取り出しの同一動作を省略
}

class Box {
	private String color; // 箱の色
private String shape;// 箱の形
private int num;// 箱の中のボールの数
static int totalNumber;

public int getNum() {
	return num;
}

public void setNum(int num) {
	this.num = num;
}

Box(String color, String shape) {
	this.color = color;
	this.shape = shape;
	this.num = 0;
}

public void insertBalls(int num) {
	this.num += num;
	totalNumber += num;
}

@Override
public String toString() {
	return "color=" + color + ", shape=" + shape + ", 個数=" + num + ">";
}
}
